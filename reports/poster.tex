\begin{posterblock}{Introduction}
\fontsize{16}{20}\selectfont
Most RL motion tracking uses \textbf{torque-driven} agents that ignore biological constraints. Musculoskeletal models are \textbf{overactuated}: each joint has multiple muscles ($>$80 actuators), making Gaussian exploration ineffective.

\vspace{6pt}
\textbf{Kinesis} [1] demonstrated RL-based motion imitation for MyoLeg but is limited to one morphology and standard exploration.

\vspace{6pt}
\textbf{DEP-RL} [2] showed that Differential Extrinsic Plasticity enables structured, body-aware exploration in overactuated systems---but without motion demonstrations.

\vspace{8pt}
\textbf{Research Question:}\\
\textit{Can DEP exploration improve RL motion tracking across diverse musculoskeletal morphologies?}

\vspace{8pt}
\textbf{Contributions:}
\begin{enumerate}[nosep, leftmargin=1.2em]
\item Integrate DEP into Kinesis PPO pipeline
\item Validate across 80, 86, 290 muscles
\item 41\% MPJPE improvement, 1.7$\times$ efficiency
\end{enumerate}
\end{posterblock}

\begin{posterblock}{Background}
\fontsize{16}{20}\selectfont
\begin{itemize}[nosep, leftmargin=1.2em]
\item \textbf{MyoSuite} [3]: MuJoCo musculoskeletal simulation with Hill-type muscle dynamics
\item \textbf{PPO}: On-policy RL ($\epsilon{=}0.2$, $\gamma{=}0.99$)
\item \textbf{MoE}: Mixture of Experts with negative mining
\item \textbf{AMASS/KIT-ML}: Reference locomotion trajectories
\end{itemize}
\end{posterblock}

\begin{posterblock}{Approach: DEP + PPO Integration}
\fontsize{16}{20}\selectfont
\textbf{Problem:} Gaussian exploration $a \sim \mathcal{N}(\mu_\theta(s), \sigma^2 I)$ treats each muscle independently. In 80--290 muscle systems, uncoordinated activations cancel out $\Rightarrow$ no meaningful movement.

\vspace{6pt}
\textbf{DEP Controller} maintains $C \in \mathbb{R}^{n \times n}$ capturing sensorimotor correlations:
\[
C_{t+1} = \textstyle\sum_{s=1}^{\tau} (M \cdot \Delta x_s) \otimes \Delta x_{s-d}^T \qquad a_{\text{DEP}} = \tanh(\kappa \cdot \hat{C} \cdot x + b)
\]

\vspace{-4pt}
where $x = \text{muscle\_len} + \alpha \cdot \text{muscle\_force}$, \; $\hat{C}$ is row-normalized, \; $\kappa{=}100$.

\vspace{6pt}
\textbf{StochSwitchDep:} With probability $p{=}0.1$, enter DEP phase of $L{=}4$ steps. Otherwise use learned policy. All transitions stored for PPO update.

\vspace{6pt}
\textbf{Imitation Reward:}
\[
r = 0.6\, e^{-200\|\mathbf{p}_{\text{ref}} - \mathbf{p}_{\text{sim}}\|^2} + 0.2\, e^{-5\|\mathbf{v}_{\text{ref}} - \mathbf{v}_{\text{sim}}\|^2} + 0.1\, r_{\text{up}} + 0.1\, r_{\text{E}}
\]

\vspace{-2pt}
\textbf{Policy:} 6-layer MLP [2048, 1536, 1024, 1024, 512, 512] with SiLU; MoE gate [1024, 512, 256].
\end{posterblock}
